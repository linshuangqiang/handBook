闭包#3

function fn(param){
    var innerFn = function(){
        return param;
    }
    param++;
    return innerFn;
}
var inner = fn(144);
//fn执行了一次，所以param只自加1次
console.log(inner())//145
console.log(inner())//145

闭包#4=>循环中的闭包

function fn(){
    var arr = [];
    for(var i = 0; i < 5; i++){
        arr[i] = function(){
            return i;
        }
    }
    return arr;
}
var arr = fn();
for(var i = 0; i < arr.length; i++){
    console.log(arr[i]);
    console.log(arr[i]())//输出5个5,即每次调用函数后都返回5
}

function fn(){
    var arr = [];
    for(var i = 0; i < 5; i++){
        arr[i] = (function(i){//1  此处的i是每次自执行是for语句的i值
            return function(){
                return i;//2  此处的i是1处的i
            }
        }(i))
    }
    return arr
}
var arr = fn();
for(var i = 0; i < arr.length; i++){
    console.log(arr[i]);
    console.log(arr[i]())
}
