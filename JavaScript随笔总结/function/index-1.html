书写方法#6（函数的祖先）

-1

Function.prototype.checkEmail = function(){
	//验证邮箱
}
var fn = function(){};
fn.checkEmail();
//or
var fn = new Function();
fn.checkEmail();

-2(抽象出可以统一添加方法的功能方法)

Function.prototype.addMethod = function(name,fn){
	this[name] = fn;
}
var methods = function(){}//or new function();
methods.addMethod('checkName',function(){
	//check user name
});
methods.addMethod('checkEmail',function(){
	//check user email
});
methods.checkName();
methods.checkEmail();

-3(针对上一个方法，支持链式)

Function.prototype.addMethod = function(name,fn){
	this[name] = fn;
	return this;
}
var methods = function(){}//or new function();
methods.addMethod('checkName',function(){
	//check user name
	return this；
}).addMethod('checkEmail',function(){
	//check user email
	return this；
})
methods.checkName().checkEmail();

书写方法#7（#6-2作对比）(类式调用方式)

Function.prototype.addMethod = function(name,fn){
	this.prototype[name] = fn;
	return this;
}
var Methods = function(){};
methods.addMethod('checkName',function(){
	//check user name
}).addMethod('checkEmail',function(){
	//check user email
})
var m = new Methods();//new 不能省略
m.checkEmail();
